73 //Serial.printf("TSPoint X: %3i, Y: %3i, Z: %3i \n", tsPoint.x, tsPoint.y, tsPoint.x);
74 TouchScreenAdapter::TouchPoint touchpoint;
75 if (tsPoint.z > (int16_t)minRawZ && tsPoint.z < (int16_t)maxRawZ) {
76 touchpoint.z = constrain(map(tsPoint.z, maxRawZ, minRawZ, 1, 255), 1, 255);
77 uint16_t x = rotation % 2 ? tsPoint.y : tsPoint.x;
78 uint16_t y = rotation % 2 ? tsPoint.x : tsPoint.y;
79 if ( rotation == 2 || rotation == 3 ) {
80 touchpoint.x = constrain(map(x, maxRawX, minRawX, 0, displayWidth), 0, displayWidth-1);
81 } else {
82 touchpoint.x = constrain(map(x, minRawX, maxRawX, 0, displayWidth), 0, displayWidth-1);
83 }
84 if ( rotation == 1 || rotation == 2 ) {
85 touchpoint.y = constrain(map(y, maxRawY, minRawY, 0, displayHeight), 0, displayHeight-1);
86 } else {
87 touchpoint.y = constrain(map(y, minRawY, maxRawY, 0, displayHeight), 0, displayHeight-1);
88 }
89 }
90 return touchpoint;
